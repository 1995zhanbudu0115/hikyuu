152 template <typename ValueType>
153 ValueType Parameter::get(const string& name) const {
154  param_map_t::const_iterator iter;
155  iter = m_params.find(name);
156  if (iter == m_params.end()) {
157  throw std::out_of_range("out_of_range in Parameter::get : " + name);
158  }
159  return boost::any_cast<ValueType>(iter->second);
160 }
