554  ar & boost::serialization::make_nvp("m_name", name);
555  m_name = UTF8ToGB(name);
556  ar & BOOST_SERIALIZATION_NVP(m_params);

562  ar & BOOST_SERIALIZATION_NVP(m_checkout_cash);
563  ar & BOOST_SERIALIZATION_NVP(m_checkin_stock);
564  ar & BOOST_SERIALIZATION_NVP(m_checkout_stock);
565  ar & BOOST_SERIALIZATION_NVP(m_borrow_cash);
566  ar & BOOST_SERIALIZATION_NVP(m_loan_list);
567  namespace bs = boost::serialization;
568  BorrowRecordList borrow;
569  ar & bs::make_nvp<BorrowRecordList>("m_borrow_stock", borrow);
570  BorrowRecordList::const_iterator bor_iter = borrow.begin();
571  for (; bor_iter != borrow.end(); ++bor_iter) {
572  m_borrow_stock[bor_iter->stock.id()] = *bor_iter;
573  }
574  PositionRecordList position;
575  ar & bs::make_nvp<PositionRecordList>("m_position", position);
576  PositionRecordList::const_iterator iter = position.begin();
577  for (; iter != position.end(); ++iter) {
578  m_position[iter->stock.id()] = *iter;
579  }
580  ar & BOOST_SERIALIZATION_NVP(m_position_history);
581 
582  position.clear();
583  ar & bs::make_nvp<PositionRecordList>("m_short_position", position);
584  iter = position.begin();
585  for (; iter != position.end(); ++iter) {
586  m_short_position[iter->stock.id()] = *iter;
587  }
588  ar & BOOST_SERIALIZATION_NVP(m_short_position_history);
589  ar & BOOST_SERIALIZATION_NVP(m_trade_list);
590  ar & BOOST_SERIALIZATION_NVP(m_actions);
591  }
